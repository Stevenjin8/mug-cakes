\begin{frame}
    \frametitle{Big Idea}
    For some unknown continuous function $f$ from a compact set $\mathcal{X} \subseteq \mathbb{R}^{K}$ to $\mathbb{R}$, we want to find
    \begin{equation*}
        \argmin_{\mathbf{x} \in \mathcal{X}} f(\mathbf{x})
    \end{equation*}
    in as few noisy samples as possible.
\end{frame}
